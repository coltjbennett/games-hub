function createRoomList() {
  const out = [];
  const tries = 220;
  for (let t = 0; t < tries && out.length < 12; t++) {
    const w = 5 + ((rng() * 5) | 0);
    const h = 5 + ((rng() * 5) | 0);

    const room = { x, y, w, h, cx: x + (w / 2) | 0, cy: y + (h / 2) | 0 };
    let bad = false;
    for (const r of out) {

        break;
      }
    }
    if (!bad) out.push(room);
  }
  out.sort((a, b) => (a.cx + a.cy) - (b.cx + b.cy));
  if (out.length < 2) return createRoomList();
  return out;
}

function carveCorridor(x1, y1, x2, y2) {
  let x = x1, y = y1;
  const horizFirst = rng() < 0.5;
  const stepX = x2 > x ? 1 : -1;
  const stepY = y2 > y ? 1 : -1;
  const carveLine = (sx, sy, ex, ey) => {
    let cx = sx, cy = sy;
    while (cx !== ex || cy !== ey) {
      map[cy][cx] = 0;
      if (cx !== ex) cx += (ex > cx ? 1 : -1);
      if (cy !== ey) cy += (ey > cy ? 1 : -1);
    }
    map[ey][ex] = 0;
  };
  if (horizFirst) {
    while (x !== x2) { map[y][x] = 0; x += stepX; }
    while (y !== y2) { map[y][x] = 0; y += stepY; }
  } else {
    while (y !== y2) { map[y][x] = 0; y += stepY; }
    while (x !== x2) { map[y][x] = 0; x += stepX; }
  }
  map[y2][x2] = 0;
}

function rectSolid(x, y, w, h) {
  const left = Math.floor(x / TILE);
  const right = Math.floor((x + w) / TILE);
  const top = Math.floor(y / TILE);
  const bottom = Math.floor((y + h) / TILE);
  for (let ty = top; ty <= bottom; ty++) {
    for (let tx = left; tx <= right; tx++) {
      if (tileSolid(tx, ty)) return true;
    }
  }
  return false;
}
function lineBlocked(x1, y1, x2, y2) {

    if (rectSolid(x - 2, y - 2, 4, 4)) return true;
  }
  return false;
}

function findPath(sx, sy, gx, gy) {
  if (tileSolid(gx, gy)) return [];
  const start = sx + ',' + sy;
  const goal = gx + ',' + gy;
  const open = [{ x: sx, y: sy, f: 0, g: 0 }];
  const came = new Map();
  const gScore = new Map([[start, 0]]);
  const closed = new Set();
  const h = (x, y) => Math.abs(x - gx) + Math.abs(y - gy);
  while (open.length) {
    open.sort((a, b) => a.f - b.f);
    const cur = open.shift();
    const key = cur.x + ',' + cur.y;
    if (key === goal) {
      const path = [[gx, gy]];
      let ck = key;
      while (came.has(ck)) {
        ck = came.get(ck);
        const [px, py] = ck.split(',').map(Number);
        path.push([px, py]);
      }
      path.reverse();
      return path;
    }
    closed.add(key);
    const nbs = [[1,0],[-1,0],[0,1],[0,-1]];
    for (const [dx, dy] of nbs) {
      const nx = cur.x + dx, ny = cur.y + dy;
      if (tileSolid(nx, ny)) continue;
      const nk = nx + ',' + ny;
      if (closed.has(nk)) continue;
      const tentative = cur.g + 1;
      if (!gScore.has(nk) || tentative < gScore.get(nk)) {
        came.set(nk, key);
        gScore.set(nk, tentative);
        const f = tentative + h(nx, ny);
        const existing = open.find(n => n.x === nx && n.y === ny);
        if (existing) existing.f = Math.min(existing.f, f);
        else open.push({ x: nx, y: ny, g: tentative, f });
      }
    }
  }
  return [];
}

    cooldown: 0,
    reloadT: 0,
  };
}
function pickupWeapon(id, ammoInMag, reserve) {
  const same = player.inv.find(s => s && s.id === id);
  if (same) {
    same.reserve += (ammoInMag || 0) + (reserve || 0);
    updateSlotsUI();
    return;
  }
  const empty = player.inv.findIndex(s => !s);
  const nw = weaponInstance(id, ammoInMag, reserve);
  if (empty !== -1) player.inv[empty] = nw;
  else {
    const old = player.inv[player.selected];
    if (old) spawnDrop(player.x, player.y, 'weapon', { weaponId: old.id, ammoInMag: old.ammoInMag, reserve: old.reserve });
    player.inv[player.selected] = nw;
  }
  updateSlotsUI();
}
function pickupAmmo(ammoType, amount) {
  for (const slot of player.inv) {
    if (slot && WEAPONS[slot.id].ammoType === ammoType) slot.reserve += amount;
  }
  updateSlotsUI();
}
function pickupMed(amount) {
  player.hp = clamp(player.hp + amount, 0, player.maxHp);
}
function getAimAssistAngle(baseAngle, originX, originY, coneDeg = 10) {
  const cone = coneDeg * Math.PI / 180;
  let bestAngle = baseAngle;
  let bestScore = Infinity;
  for (const enemy of enemies) {
    if (!enemy || enemy.hp <= 0) continue;
    const dx = enemy.x - originX;
    const dy = enemy.y - originY;
    const d = Math.hypot(dx, dy);
    if (d > 980) continue;
    const ang = Math.atan2(dy, dx);
    let diff = ang - baseAngle;
    diff = Math.atan2(Math.sin(diff), Math.cos(diff));
    const abs = Math.abs(diff);
    if (abs <= cone) {
      const score = abs * 10 + d / 1000;
      if (score < bestScore) {
        bestScore = score;
        bestAngle = ang;
      }
    }
  }
  return bestAngle;
}

function shootFromSlot(slot, owner) {
  const def = WEAPONS[slot.id];
  if (slot.ammoInMag <= 0 || slot.reloadT > 0 || slot.cooldown > 0) return false;
  slot.cooldown = 1 / def.fireRate;
  slot.ammoInMag--;
  const originX = owner.x + Math.cos(owner.angle) * 20;
  const originY = owner.y + Math.sin(owner.angle) * 20;
  const baseAngle = owner === player ? getAimAssistAngle(owner.angle, originX, originY, 10) : owner.angle;
  for (let i = 0; i < def.pellets; i++) {
    const spread = (Math.random() - 0.5) * def.spread;
    spawnBullet(originX, originY, baseAngle + spread, def.speed, def.damage, owner === player ? 'player' : 'enemy', owner === player ? '#eef6ff' : '#ffcbcb');
  }
  shake = Math.min(10, shake + 1.4);
  burst(originX, originY, owner === player ? '#b8d8ff' : '#ffb9b9', 4, 120);
  return true;
}
function reloadSlot(slot) {
  const def = WEAPONS[slot.id];
  if (slot.reloadT > 0) return;
  if (slot.ammoInMag >= def.magSize || slot.reserve <= 0) return;
  slot.reloadT = def.reload;
}
function finishReload(slot) {
  const def = WEAPONS[slot.id];
  const need = def.magSize - slot.ammoInMag;
  const take = Math.min(need, slot.reserve);
  slot.ammoInMag += take;
  slot.reserve -= take;
}

function generateLevel(lv) {
  seed = (lv * 9973 + 12345) >>> 0;
  rng = mulberry32(seed);
  map = Array.from({ length: MAP_H }, () => Array(MAP_W).fill(1));
  rooms = createRoomList();
  for (const room of rooms) carveRoom(room);
  for (let i = 1; i < rooms.length; i++) {
    carveCorridor(rooms[i - 1].cx, rooms[i - 1].cy, rooms[i].cx, rooms[i].cy);
  }
  const start = rooms[0];
  const end = rooms[rooms.length - 1];
  exitTile = { x: end.cx, y: end.cy, open: false };
  player.x = start.cx * TILE + TILE / 2;
  player.y = start.cy * TILE + TILE / 2;
  player.vx = player.vy = 0;
  player.angle = 0;
  player.moveX = player.moveY = 0;
  player.aimX = player.aimY = 0;
  player.fireHeld = false;
  player.reloadHeld = false;
  if (!player.inv.length) {
    player.inv = [weaponInstance('pistol'), null, null, null];
    player.selected = 0;
  }
  for (const s of player.inv) {
    if (s) {
      s.cooldown = 0;
      s.reloadT = 0;
    }
  }
  clearWorld();

  const enemyCount = 8 + lv * 4;
  const types = ['grunt', 'gunner', 'grunt', 'gunner', 'tank'];
  for (let i = 0; i < enemyCount; i++) {
    const room = rooms[1 + ((rng() * (rooms.length - 1)) | 0)];
    const type = types[((rng() * types.length) | 0)];
    const x = (room.x + 1 + ((rng() * Math.max(1, room.w - 2)) | 0)) * TILE + TILE / 2;
    const y = (room.y + 1 + ((rng() * Math.max(1, room.h - 2)) | 0)) * TILE + TILE / 2;
    enemies.push(spawnEnemy(type, x, y));
  }

  const ammoDrops = 3 + (lv >> 1);
  for (let i = 0; i < ammoDrops; i++) {
    const room = randChoice(rooms.slice(1));
    const tx = room.x + 1 + ((rng() * Math.max(1, room.w - 2)) | 0);
    const ty = room.y + 1 + ((rng() * Math.max(1, room.h - 2)) | 0);
    const kinds = ['light', 'shell', 'rifle'];
    const kind = kinds[((rng() * kinds.length) | 0)];
    spawnDrop(tx * TILE + TILE / 2, ty * TILE + TILE / 2, 'ammo', { ammoType: kind, amount: AMMO_DROP[kind] });
  }
  if (lv % 2 === 0) {
    const room = randChoice(rooms.slice(1));
    spawnDrop((room.x + 2) * TILE, (room.y + 2) * TILE, 'med');
  }
  updateSlotsUI();
  gameOver = false;
  hideOverlay();
}

function enemyShoot(enemy, ang, spread = 0.12, pellets = 1, damage = 8, speed = 880) {
  const ox = enemy.x + Math.cos(ang) * 18;
  const oy = enemy.y + Math.sin(ang) * 18;
  for (let i = 0; i < pellets; i++) {
    const a = ang + (Math.random() - 0.5) * spread;
    spawnBullet(ox, oy, a, speed, damage, 'enemy', '#ffbbbb');
  }
  burst(ox, oy, '#ff9d9d', 3, 100);
}

function updateEnemy(enemy, dt) {
  if (enemy.hp <= 0) return;
  enemy.shootCD = Math.max(0, enemy.shootCD - dt);
  enemy.pathTimer -= dt;
  const dx = player.x - enemy.x;
  const dy = player.y - enemy.y;
  const d = Math.hypot(dx, dy);
  const sees = d < enemy.sight && !lineBlocked(enemy.x, enemy.y, player.x, player.y);
  enemy.angle = Math.atan2(dy, dx);

  const radius = 15;
  if (enemy.type === 'grunt') {
    if (sees) {
      if (d < 60) {
        if (enemy.shootCD <= 0) { damagePlayer(9); enemy.shootCD = 0.9; }
      } else {
        moveEntity(enemy, Math.cos(enemy.angle) * enemy.speed * dt, Math.sin(enemy.angle) * enemy.speed * dt, radius);
        if (enemy.shootCD <= 0 && d < 340) {
          enemyShoot(enemy, enemy.angle, 0.06, 1, 10, 900);
          enemy.shootCD = 0.75;
        }
      }
    } else {
      followPath(enemy, dt, radius, 1.0);
    }
  } else if (enemy.type === 'gunner') {
    if (sees) {
      const want = d > 260 ? 1 : d < 180 ? -1 : 0;
      moveEntity(enemy, Math.cos(enemy.angle) * enemy.speed * dt * want, Math.sin(enemy.angle) * enemy.speed * dt * want, radius);
      if (enemy.shootCD <= 0 && d < 620) {
        enemyShoot(enemy, enemy.angle, 0.11, 1, 8, 1020);
        enemy.shootCD = 0.14;
      }
    } else {
      followPath(enemy, dt, radius, 1.15);
    }
  } else {
    if (sees) {
      const want = d > 150 ? 1 : 0;
      moveEntity(enemy, Math.cos(enemy.angle) * enemy.speed * dt * want, Math.sin(enemy.angle) * enemy.speed * dt * want, radius + 2);
      if (enemy.shootCD <= 0 && d < 220) {
        enemyShoot(enemy, enemy.angle, 0.28, 6, 7, 820);
        enemy.shootCD = 1.15;
      }
    } else {
      followPath(enemy, dt, radius + 2, 1.35);
    }
  }

  if (d < 28) {
    enemy.contact += dt;
    if (enemy.contact > 0.6) {
      damagePlayer(enemy.type === 'tank' ? 18 : 8);
      enemy.contact = 0;
    }
  } else enemy.contact = 0;
}

function moveEntity(ent, dx, dy, radius) {
  const tryX = ent.x + dx;
  if (!rectSolid(tryX - radius, ent.y - radius, radius * 2, radius * 2)) ent.x = tryX;
  else {
    const step = Math.sign(dx);
    if (step) {
      for (let i = 0; i < Math.abs(dx); i += 2) {
        const nx = ent.x + Math.sign(dx) * Math.min(2, Math.abs(dx) - i);
        if (!rectSolid(nx - radius, ent.y - radius, radius * 2, radius * 2)) ent.x = nx;
        else break;
      }
    }
  }
  const tryY = ent.y + dy;
  if (!rectSolid(ent.x - radius, tryY - radius, radius * 2, radius * 2)) ent.y = tryY;
  else {
    const step = Math.sign(dy);
    if (step) {
      for (let i = 0; i < Math.abs(dy); i += 2) {
        const ny = ent.y + Math.sign(dy) * Math.min(2, Math.abs(dy) - i);
        if (!rectSolid(ent.x - radius, ny - radius, radius * 2, radius * 2)) ent.y = ny;
        else break;
      }
    }
  }
}

function followPath(enemy, dt, radius, repathFactor) {
  if (enemy.pathTimer <= 0 || !enemy.path.length || enemy.pathIndex >= enemy.path.length) {
    const sx = (enemy.x / TILE) | 0;
    const sy = (enemy.y / TILE) | 0;
    const gx = (player.x / TILE) | 0;
    const gy = (player.y / TILE) | 0;
    enemy.path = findPath(sx, sy, gx, gy);
    enemy.pathIndex = 0;
    enemy.pathTimer = 0.5 + repathFactor * 0.3;
  }
  const next = enemy.path[enemy.pathIndex];
  if (!next) return;
  const [tx, ty] = tileCenter(next[0], next[1]);
  const dx = tx - enemy.x, dy = ty - enemy.y;
  const d = Math.hypot(dx, dy);
  if (d < 10 && enemy.pathIndex < enemy.path.length - 1) enemy.pathIndex++;
  else moveEntity(enemy, (dx / (d || 1)) * enemy.speed * dt, (dy / (d || 1)) * enemy.speed * dt, radius);
}

function updatePlayer(dt) {
  for (const slot of player.inv) {
    if (!slot) continue;
    if (slot.cooldown > 0) slot.cooldown = Math.max(0, slot.cooldown - dt);
    if (slot.reloadT > 0) {
      slot.reloadT -= dt;
      if (slot.reloadT <= 0) finishReload(slot);
    }
  }

  const kbX = (keys.has('KeyD') || keys.has('ArrowRight') ? 1 : 0) - (keys.has('KeyA') || keys.has('ArrowLeft') ? 1 : 0);
  const kbY = (keys.has('KeyS') || keys.has('ArrowDown') ? 1 : 0) - (keys.has('KeyW') || keys.has('ArrowUp') ? 1 : 0);
  const moveX = (player.moveX || kbX) + 0;
  const moveY = (player.moveY || kbY) + 0;
  const [mx, my, ml] = norm(moveX, moveY);
  const accel = 950;
  const friction = 8.2;
  player.vx += mx * accel * dt;
  player.vy += my * accel * dt;
  if (!ml && !kbX && !kbY) {
    player.vx *= Math.max(0, 1 - friction * dt);
    player.vy *= Math.max(0, 1 - friction * dt);
  }
  const maxSpeed = 190;
  const speed = Math.hypot(player.vx, player.vy);
  if (speed > maxSpeed) {
    player.vx = (player.vx / speed) * maxSpeed;
    player.vy = (player.vy / speed) * maxSpeed;
  }
  moveEntity(player, player.vx * dt, player.vy * dt, 15);

  const aimActive = player.aimX !== 0 || player.aimY !== 0;
  if (aimActive) player.angle = Math.atan2(player.aimY, player.aimX);
  else player.angle = Math.atan2(mouse.y - player.y, mouse.x - player.x);

  const slot = getSelectedWeapon();
  if (slot) {
    const def = WEAPONS[slot.id];
    const fireKey = keys.has('Space') || mouse.down || player.fireHeld;
    if (fireKey && slot.cooldown <= 0 && slot.reloadT <= 0) shootFromSlot(slot, player);
    if (player.reloadHeld || keys.has('KeyR')) reloadSlot(slot);
  }

  if (rectSolid(player.x - 12, player.y - 12, 24, 24)) {
    // tiny nudge if spawned in a bad spot
    player.x += 2; player.y += 2;
  }
}

function updateBullets(dt) {
  for (let i = bullets.length - 1; i >= 0; i--) {
    const b = bullets[i];
    const steps = Math.max(1, Math.ceil(Math.hypot(b.vx * dt, b.vy * dt) / 12));
    let hit = false;
    for (let s = 0; s < steps; s++) {
      b.x += b.vx * dt / steps;
      b.y += b.vy * dt / steps;
      if (rectSolid(b.x - 2, b.y - 2, 4, 4)) { hit = true; break; }
      if (b.owner === 'player') {
        for (const e of enemies) {
          if (e.hp > 0 && dist(b.x, b.y, e.x, e.y) < 16) {
            e.hp -= b.damage;
            hit = true;
            shake = Math.min(8, shake + 0.35);
            burst(b.x, b.y, '#ffd8d8', 2, 80);
            if (e.hp <= 0) onEnemyKilled(e);
            break;
          }
        }
      } else {
        if (dist(b.x, b.y, player.x, player.y) < 16) {
          damagePlayer(b.damage);
          hit = true;
          break;
        }
      }
      if (hit) break;
    }
    b.life -= dt;
    if (hit || b.life <= 0) bullets.splice(i, 1);
  }
}

function onEnemyKilled(enemy) {
  score += enemy.type === 'tank' ? 30 : enemy.type === 'gunner' ? 18 : 12;
  burst(enemy.x, enemy.y, enemy.type === 'tank' ? '#ff9288' : '#c7d7ff', 12, 180);
  const r = Math.random();
  if (r < 0.4) spawnDrop(enemy.x, enemy.y, 'ammo', { ammoType: WEAPONS[enemy.weaponId].ammoType, amount: AMMO_DROP[WEAPONS[enemy.weaponId].ammoType] });
  if (enemy.type === 'gunner' && Math.random() < 0.45) spawnDrop(enemy.x, enemy.y, 'weapon', { weaponId: 'smg', ammoInMag: 8 + ((Math.random() * 8) | 0), reserve: 24 + ((Math.random() * 24) | 0) });
  if (enemy.type === 'grunt' && Math.random() < 0.18) spawnDrop(enemy.x, enemy.y, 'weapon', { weaponId: 'pistol', ammoInMag: 6 + ((Math.random() * 4) | 0), reserve: 12 + ((Math.random() * 18) | 0) });
  if (enemy.type === 'tank' && Math.random() < 0.5) spawnDrop(enemy.x, enemy.y, 'med');
}

function updateDrops(dt) {
  for (let i = drops.length - 1; i >= 0; i--) {
    const d = drops[i];
    d.bob += dt * 4;
    if (dist(player.x, player.y, d.x, d.y) < 22) {
      if (d.kind === 'weapon') pickupWeapon(d.weaponId, d.ammoInMag, d.reserve);
      if (d.kind === 'ammo') pickupAmmo(d.ammoType, d.amount);
      if (d.kind === 'med') pickupMed(25);
      drops.splice(i, 1);
    }
  }
}

function updateParticles(dt) {
  for (let i = particles.length - 1; i >= 0; i--) {
    const p = particles[i];
    p.t += dt;
    p.x += p.vx * dt;
    p.y += p.vy * dt;
    p.vx *= Math.max(0, 1 - 6 * dt);
    p.vy *= Math.max(0, 1 - 6 * dt);
    if (p.t >= p.life) particles.splice(i, 1);
  }
}

function updateLevelState() {
  if (enemies.every(e => e.hp <= 0)) exitTile.open = true;

  const pty = (player.y / TILE) | 0;
  if (exitTile.open && ptx === exitTile.x && pty === exitTile.y) {

    generateLevel(level);
  }
}

function drawWorld() {
  ctx.save();
  ctx.clearRect(0, 0, W, H);
  const sx = player.x - W / 2;
  const sy = player.y - H / 2;
  camX = sx;
  camY = sy;
  let shakeX = 0, shakeY = 0;
  if (shake > 0.01) {
    shakeX = (Math.random() - 0.5) * shake;
    shakeY = (Math.random() - 0.5) * shake;
    shake *= 0.92;
  } else shake = 0;
  ctx.translate(-sx + shakeX, -sy + shakeY);

  const minTX = clamp(((sx - TILE) / TILE) | 0, 0, MAP_W - 1);
  const maxTX = clamp(((sx + W + TILE) / TILE) | 0, 0, MAP_W - 1);
  const minTY = clamp(((sy - TILE) / TILE) | 0, 0, MAP_H - 1);
  const maxTY = clamp(((sy + H + TILE) / TILE) | 0, 0, MAP_H - 1);

  for (let ty = minTY; ty <= maxTY; ty++) {
    for (let tx = minTX; tx <= maxTX; tx++) {
      const x = tx * TILE, y = ty * TILE;
      if (map[ty][tx] === 0) drawFloor(tx, ty, x, y);
      else drawWall(x, y);
    }
  }

  if (exitTile.x >= 0) drawExit(exitTile.x * TILE, exitTile.y * TILE);

  for (const d of drops) {
    ctx.save();
    ctx.translate(d.x, d.y + Math.sin(d.bob) * 3);
    if (d.kind === 'weapon') {
      const sprite = SPRITES[d.weaponId];
      ctx.drawImage(sprite, -24, -24, 48, 48);
    } else if (d.kind === 'ammo') {
      const sprite = SPRITES[d.ammoType === 'rifle' ? 'rifleAmmo' : d.ammoType];
      ctx.drawImage(sprite, -18, -18, 36, 36);
    } else {
      ctx.drawImage(SPRITES.medkit, -18, -18, 36, 36);
    }
    ctx.restore();
  }

  for (const p of particles) {
    const a = 1 - p.t / p.life;
    ctx.fillStyle = p.color;
    ctx.globalAlpha = a;
    ctx.fillRect(p.x - 1.5, p.y - 1.5, 3, 3);
    ctx.globalAlpha = 1;
  }

  for (const b of bullets) {
    ctx.fillStyle = b.color;
    ctx.beginPath();
    ctx.arc(b.x, b.y, b.owner === 'player' ? 2.2 : 2.6, 0, TAU);
    ctx.fill();
  }

  for (const e of enemies) {
    if (e.hp <= 0) continue;
    const s = e.type === 'tank' ? 1.12 : 1;
    drawEntitySprite(SPRITES[e.type], e.x, e.y, e.angle, s);
    // health bar
    ctx.fillStyle = 'rgba(0,0,0,0.45)';
    ctx.fillRect(e.x - 18, e.y - 34, 36, 5);
    ctx.fillStyle = e.type === 'tank' ? '#ff7b70' : '#7be2ff';
    ctx.fillRect(e.x - 18, e.y - 34, 36 * (e.hp / e.maxHp), 5);
  }

  drawEntitySprite(SPRITES.player, player.x, player.y, player.angle, 1);

  if (mouse.down || player.fireHeld) {
    ctx.strokeStyle = 'rgba(140,180,255,0.5)';
    ctx.beginPath();
    ctx.moveTo(player.x, player.y);
    ctx.lineTo(player.x + Math.cos(player.angle) * 42, player.y + Math.sin(player.angle) * 42);
    ctx.stroke();
  }

  ctx.restore();
}

function loop(now) {
  if (!loop.last) loop.last = now;
  const dt = Math.min((now - loop.last) / 1000, 0.033);
  loop.last = now;

  if (!gameOver) {
    updatePlayer(dt);
    for (const e of enemies) updateEnemy(e, dt);
    updateBullets(dt);
    updateDrops(dt);
    updateParticles(dt);
    updateLevelState();
  } else {
    updateParticles(dt);
  }

  drawWorld();
  updateHUD();
  requestAnimationFrame(loop);
}

function stickControl(stickEl, nubEl, onMove, onEnd) {
  const state = { id: null, cx: 0, cy: 0, active: false };
  const max = 42;

  const reset = () => {
    if (!state.active) return;
    state.active = false;
    state.id = null;
    nubEl.style.transform = 'translate(0px, 0px)';
    onEnd();
  };

  stickEl.addEventListener('pointerdown', e => {
    e.preventDefault();
    state.id = e.pointerId;
    const rect = stickEl.getBoundingClientRect();
    state.cx = rect.left + rect.width / 2;
    state.cy = rect.top + rect.height / 2;
    state.active = true;
    try { stickEl.setPointerCapture(e.pointerId); } catch (_) {}
    update(e.clientX, e.clientY);
  });

  const moveHandler = e => {
    if (!state.active || e.pointerId !== state.id) return;
    update(e.clientX, e.clientY);
  };
  const endHandler = e => {
    if (e && e.pointerId !== state.id) return;
    reset();
  };

  stickEl.addEventListener('pointermove', moveHandler);
  stickEl.addEventListener('pointerup', endHandler);
  stickEl.addEventListener('pointercancel', endHandler);
  stickEl.addEventListener('lostpointercapture', reset);
  window.addEventListener('pointerup', endHandler);
  window.addEventListener('pointercancel', endHandler);

  function update(x, y) {
    const dx = x - state.cx;
    const dy = y - state.cy;
    const l = Math.hypot(dx, dy);
    const nx = l > max ? dx / l * max : dx;
    const ny = l > max ? dy / l * max : dy;
    nubEl.style.transform = `translate(${nx}px, ${ny}px)`;
    onMove(nx / max, ny / max, Math.min(1, l / max));
  }
}

function generateLevel(lv) {
  seed = (lv * 9779 + 13579) >>> 0;
  rng = mulberry32(seed);
  map = Array.from({ length: MAP_H }, () => Array(MAP_W).fill(1));
  props = [];
  bullets = [];
  enemies = [];
  drops = [];
  particles = [];

  const spacing = 9;
  const roadW = 2;
  const sidewalkW = 1;

  for (let y = 0; y < MAP_H; y++) {
    for (let x = 0; x < MAP_W; x++) {
      const mx = x % spacing;
      const my = y % spacing;
      const blockX = (x / spacing) | 0;
      const blockY = (y / spacing) | 0;
      const park = (((blockX * 31 + blockY * 17 + lv * 7) % 6) === 0);
      if (mx < roadW || my < roadW) map[y][x] = 0;
      else if (mx === roadW || my === roadW) map[y][x] = 2;
      else map[y][x] = park ? 3 : 1;
    }
  }

  const openTiles = [];
  for (let y = 0; y < MAP_H; y++) {
    for (let x = 0; x < MAP_W; x++) {
      if (!tileSolid(x, y)) openTiles.push({ x, y });
      if ((map[y][x] === 2 || map[y][x] === 3) && rng() < 0.07) {
        const types = ['trash', 'grate', 'hydrant', 'cone', 'lamp', 'palm'];
        const t = types[(rng() * types.length) | 0];
        props.push({ type: t, x: x * TILE + TILE / 2, y: y * TILE + TILE / 2, rot: rng() * TAU });
      }
    }
  }

  const startTile = openTiles.find(t => t.x < 4 && t.y < 4) || openTiles[0];
  let far = openTiles[openTiles.length - 1] || openTiles[0];
  let best = -1;
  for (const t of openTiles) {
    const d = (t.x - startTile.x) ** 2 + (t.y - startTile.y) ** 2;
    if (d > best) { best = d; far = t; }
  }

  exitTile = { x: far.x, y: far.y, open: false };
  player.x = startTile.x * TILE + TILE / 2;
  player.y = startTile.y * TILE + TILE / 2;
  player.vx = player.vy = 0;
  player.angle = 0;
  player.moveX = player.moveY = 0;
  player.aimX = player.aimY = 0;
  player.fireHeld = false;
  player.reloadHeld = false;

  if (!player.inv.length) {
    player.inv = [weaponInstance('pistol'), null, null, null];
    player.selected = 0;
  }
  for (const s of player.inv) {
    if (s) {
      s.cooldown = 0;
      s.reloadT = 0;
    }
  }

  const enemyCount = 7 + lv * 3;
  for (let i = 0; i < enemyCount; i++) {
    const t = openTiles[((rng() * openTiles.length) | 0)];
    if (!t) continue;
    if (Math.hypot(t.x - startTile.x, t.y - startTile.y) < 6) continue;
    const typeRoll = rng();
    const type = typeRoll < 0.55 ? 'grunt' : typeRoll < 0.88 ? 'gunner' : 'tank';
    const ex = t.x * TILE + TILE / 2 + (rng() - 0.5) * 12;
    const ey = t.y * TILE + TILE / 2 + (rng() - 0.5) * 12;
    enemies.push(spawnEnemy(type, ex, ey));
  }

  const weaponChance = lv >= 2 ? 0.42 : 0.22;
  for (let i = 0; i < 3 + (lv >> 1); i++) {
    const t = openTiles[((rng() * openTiles.length) | 0)];
    if (!t) continue;
    const kinds = ['light', 'shell', 'rifle'];
    const kind = kinds[(rng() * kinds.length) | 0];
    spawnDrop(t.x * TILE + TILE / 2, t.y * TILE + TILE / 2, 'ammo', { ammoType: kind, amount: AMMO_DROP[kind] });
  }
  if (rng() < weaponChance) {
    const t = openTiles[((rng() * openTiles.length) | 0)];
    if (t) {
      const loot = ['smg', 'shotgun', 'rifle'][((rng() * 3) | 0)];
      spawnDrop(t.x * TILE + TILE / 2, t.y * TILE + TILE / 2, 'weapon', {
        weaponId: loot,
        ammoInMag: Math.max(1, ((WEAPONS[loot].magSize * 0.5) | 0)),
        reserve: Math.max(0, ((WEAPONS[loot].reserve * 0.35) | 0))
      });
    }
  }
  if (rng() < 0.65) {
    const t = openTiles[((rng() * openTiles.length) | 0)];
    if (t) spawnDrop(t.x * TILE + TILE / 2, t.y * TILE + TILE / 2, 'med');
  }

  updateSlotsUI();
  gameOver = false;
  hideOverlay();
}

function updateEnemy(enemy, dt) {
  if (enemy.hp <= 0) return;
  enemy.shootCD = Math.max(0, enemy.shootCD - dt);
  enemy.pathTimer -= dt;
  enemy.strafeTimer -= dt;

  const dx = player.x - enemy.x;
  const dy = player.y - enemy.y;
  const d = Math.hypot(dx, dy);
  const sees = d < enemy.sight && !lineBlocked(enemy.x, enemy.y, player.x, player.y);
  enemy.angle = Math.atan2(dy, dx);

  const radius = enemy.type === 'tank' ? 17 : 14;

  if (sees) {
    const dirX = dx / (d || 1);
    const dirY = dy / (d || 1);
    const strafe = enemy.strafeTimer > 0 ? enemy.strafeDir : (enemy.strafeDir *= -1, enemy.strafeTimer = 0.7 + Math.random() * 0.6, enemy.strafeDir);
    const wantedDist = enemy.type === 'grunt' ? 180 : enemy.type === 'gunner' ? 260 : 170;
    const chase = d > wantedDist ? 1 : d < wantedDist * 0.7 ? -0.6 : 0.2;
    const side = strafe * 0.42;
    moveEntity(enemy,
      (dirX * chase - dirY * side) * enemy.speed * dt,
      (dirY * chase + dirX * side) * enemy.speed * dt,
      radius
    );

    if (enemy.type === 'grunt') {
      if (enemy.shootCD <= 0 && d < 300) {
        enemyShoot(enemy, enemy.angle, 0.045, 1, 4, 800);
        enemy.shootCD = 1.65;
      }
    } else if (enemy.type === 'gunner') {
      if (enemy.shootCD <= 0 && d < 470) {
        enemyShoot(enemy, enemy.angle, 0.09, 1, 3, 900);
        enemy.shootCD = 0.72;
      }
    } else {
      if (enemy.shootCD <= 0 && d < 210) {
        enemyShoot(enemy, enemy.angle, 0.18, 4, 5, 720);
        enemy.shootCD = 1.8;
      }
    }
  } else {
    followPath(enemy, dt, radius, enemy.type === 'gunner' ? 1.3 : 1.0);
  }

  if (d < 28) {
    enemy.contact += dt;
    if (enemy.contact > 0.85) {
      damagePlayer(enemy.type === 'tank' ? 12 : 6);
      enemy.contact = 0;
    }
  } else {
    enemy.contact = 0;
  }
}

function updatePlayer(dt) {
  for (const slot of player.inv) {
    if (!slot) continue;
    if (slot.cooldown > 0) slot.cooldown = Math.max(0, slot.cooldown - dt);
    if (slot.reloadT > 0) {
      slot.reloadT -= dt;
      if (slot.reloadT <= 0) finishReload(slot);
    }
  }

  const kbX = (keys.has('KeyD') || keys.has('ArrowRight') ? 1 : 0) - (keys.has('KeyA') || keys.has('ArrowLeft') ? 1 : 0);
  const kbY = (keys.has('KeyS') || keys.has('ArrowDown') ? 1 : 0) - (keys.has('KeyW') || keys.has('ArrowUp') ? 1 : 0);
  const moveX = (player.moveX || 0) + kbX;
  const moveY = (player.moveY || 0) + kbY;
  const moveMag = Math.hypot(moveX, moveY);
  const mx = moveMag ? moveX / moveMag : 0;
  const my = moveMag ? moveY / moveMag : 0;
  const accel = 980 * Math.min(1, moveMag);
  const friction = 8.8;
  player.vx += mx * accel * dt;
  player.vy += my * accel * dt;
  if (!moveMag) {
    player.vx *= Math.max(0, 1 - friction * dt);
    player.vy *= Math.max(0, 1 - friction * dt);
  }
  const maxSpeed = 220;
  const speed = Math.hypot(player.vx, player.vy);
  if (speed > maxSpeed) {
    player.vx = (player.vx / speed) * maxSpeed;
    player.vy = (player.vy / speed) * maxSpeed;
  }
  moveEntity(player, player.vx * dt, player.vy * dt, 15);

  const aimActive = player.aimX !== 0 || player.aimY !== 0;
  if (aimActive) player.angle = Math.atan2(player.aimY, player.aimX);

  const slot = getSelectedWeapon();
  if (slot) {
    const autoFire = aimActive || mouse.down || player.fireHeld || keys.has('Space');
    if (slot.ammoInMag <= 0 && slot.reserve > 0 && slot.reloadT <= 0) {
      reloadSlot(slot);
    }
    if (autoFire && slot.reloadT <= 0 && slot.cooldown <= 0 && slot.ammoInMag > 0) {
      shootFromSlot(slot, player);
    }
  }

  updateSlotsUI();
}

function drawWorld() {
  ctx.save();
  ctx.clearRect(0, 0, W, H);
  ctx.fillStyle = '#8fd0ff';
  ctx.fillRect(0, 0, W, H);

  const sx = player.x - W / 2;
  const sy = player.y - H / 2;
  camX = sx;
  camY = sy;

  let shakeX = 0, shakeY = 0;
  if (shake > 0.01) {
    shakeX = (Math.random() - 0.5) * shake;
    shakeY = (Math.random() - 0.5) * shake;
    shake *= 0.92;
  } else {
    shake = 0;
  }
  ctx.translate(-sx + shakeX, -sy + shakeY);

  const minTX = clamp(((sx - TILE) / TILE) | 0, 0, MAP_W - 1);
  const maxTX = clamp(((sx + W + TILE) / TILE) | 0, 0, MAP_W - 1);
  const minTY = clamp(((sy - TILE) / TILE) | 0, 0, MAP_H - 1);
  const maxTY = clamp(((sy + H + TILE) / TILE) | 0, 0, MAP_H - 1);

  ctx.fillStyle = 'rgba(255,255,255,0.07)';
  ctx.fillRect(sx - 200, sy - 200, W + 400, 180);

  for (let ty = minTY; ty <= maxTY; ty++) {
    for (let tx = minTX; tx <= maxTX; tx++) {
      const x = tx * TILE, y = ty * TILE;
      if (map[ty][tx] === 1) drawWall(x, y, tx, ty);
      else drawFloor(tx, ty, x, y);
    }
  }

  for (const prop of props) {
    if (prop.x < sx - 60 || prop.x > sx + W + 60 || prop.y < sy - 60 || prop.y > sy + H + 60) continue;
    drawProp(prop);
  }

  if (exitTile.x >= 0) drawExit(exitTile.x * TILE, exitTile.y * TILE);

  for (const d of drops) {
    ctx.save();
    ctx.translate(d.x, d.y + Math.sin(d.bob) * 3);
    if (d.kind === 'weapon') {
      ctx.drawImage(SPRITES[d.weaponId], -24, -24, 48, 48);
    } else if (d.kind === 'ammo') {
      const sprite = SPRITES[d.ammoType === 'rifle' ? 'rifleAmmo' : d.ammoType];
      ctx.drawImage(sprite, -18, -18, 36, 36);
    } else {
      ctx.drawImage(SPRITES.medkit, -18, -18, 36, 36);
    }
    ctx.restore();
  }

  for (const p of particles) {
    const a = 1 - p.t / p.life;
    ctx.fillStyle = p.color;
    ctx.globalAlpha = a;
    ctx.fillRect(p.x - 1.5, p.y - 1.5, 3, 3);
    ctx.globalAlpha = 1;
  }

  for (const b of bullets) {
    ctx.fillStyle = b.color;
    ctx.beginPath();
    ctx.arc(b.x, b.y, b.owner === 'player' ? 2.2 : 2.5, 0, TAU);
    ctx.fill();
  }

  for (const e of enemies) {
    if (e.hp <= 0) continue;
    const s = e.type === 'tank' ? 1.08 : 1;
    drawEntitySprite(SPRITES[e.type], e.x, e.y, e.angle, s);
    ctx.fillStyle = 'rgba(0,0,0,0.35)';
    ctx.fillRect(e.x - 18, e.y - 34, 36, 5);
    ctx.fillStyle = e.type === 'tank' ? '#ff8a73' : '#7be2ff';
    ctx.fillRect(e.x - 18, e.y - 34, 36 * (e.hp / e.maxHp), 5);
  }

  drawEntitySprite(SPRITES.player, player.x, player.y, player.angle, 1);

  ctx.restore();
}

function onEnemyKilled(enemy) {
  score += enemy.type === 'tank' ? 30 : enemy.type === 'gunner' ? 18 : 12;
  burst(enemy.x, enemy.y, enemy.type === 'tank' ? '#ff9288' : '#c7d7ff', 12, 180);

  const ammoType = WEAPONS[enemy.weaponId].ammoType;
  const ammoAmount = Math.max(AMMO_DROP[ammoType], Math.round(AMMO_DROP[ammoType] * 1.5));
  spawnDrop(enemy.x, enemy.y, 'ammo', { ammoType, amount: ammoAmount });

  const weaponDropChance = enemy.type === 'tank' ? 0.7 : enemy.type === 'gunner' ? 0.6 : 0.5;
  if (Math.random() < weaponDropChance) {
    const ammoInMag = Math.max(1, Math.round(WEAPONS[enemy.weaponId].magSize * 0.5));
    const reserve = Math.max(0, Math.round(WEAPONS[enemy.weaponId].reserve * 0.35));
    spawnDrop(enemy.x, enemy.y, 'weapon', { weaponId: enemy.weaponId, ammoInMag, reserve });
  }

  if (enemy.type === 'tank' && Math.random() < 0.35) spawnDrop(enemy.x, enemy.y, 'med');
}

function stickControl(stickEl, nubEl, onMove, onEnd) {
  const state = { id: null, cx: 0, cy: 0, active: false };
  const max = 42;

  const reset = () => {
    if (!state.active) return;
    state.active = false;
    state.id = null;
    nubEl.style.transform = 'translate(0px, 0px)';
    onEnd();
  };

  stickEl.addEventListener('pointerdown', e => {
    e.preventDefault();
    state.id = e.pointerId;
    const rect = stickEl.getBoundingClientRect();
    state.cx = rect.left + rect.width / 2;
    state.cy = rect.top + rect.height / 2;
    state.active = true;
    try { stickEl.setPointerCapture(e.pointerId); } catch (_) {}
    update(e.clientX, e.clientY);
  });

  const moveHandler = e => {
    if (!state.active || e.pointerId !== state.id) return;
    update(e.clientX, e.clientY);
  };
  const endHandler = e => {
    if (e && e.pointerId !== state.id) return;
    reset();
  };

  stickEl.addEventListener('pointermove', moveHandler);
  stickEl.addEventListener('pointerup', endHandler);
  stickEl.addEventListener('pointercancel', endHandler);
  stickEl.addEventListener('lostpointercapture', reset);
  window.addEventListener('pointerup', endHandler);
  window.addEventListener('pointercancel', endHandler);

  function update(x, y) {
    const dx = x - state.cx;
    const dy = y - state.cy;
    const l = Math.hypot(dx, dy);
    const nx = l > max ? dx / l * max : dx;
    const ny = l > max ? dy / l * max : dy;
    nubEl.style.transform = `translate(${nx}px, ${ny}px)`;
    onMove(nx / max, ny / max, Math.min(1, l / max));
  }
}
