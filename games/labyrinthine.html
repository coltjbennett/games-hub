<script>
    // Force-blocks double-tap zoom using event listeners
    let lastTouchEnd = 0;
    document.addEventListener('touchend', function (event) {
        let now = (new Date()).getTime();
        if (now - lastTouchEnd <= 300) {
            event.preventDefault(); // This blocks the browser's automatic zoom
        }
        lastTouchEnd = now;
    }, false);
</script>
    <script type="module">
        import * as THREE from 'https://unpkg.com/three@0.160.0/build/three.module.js';

        // --- GAME STATE ---
        const gameState = {
            health: 100,
            battery: 100,
            flashOn: true,
            ammo: 6,
            maxAmmo: 6,
            isReloading: false,
            isADS: false,
            weaponIndex: 0,
            weapons: [
                { name: "Revolver", color: 0x222222, rate: 600, damage: 45, auto: false, max: 6 },
                { name: "Rusty Pipe", color: 0x4a3b2c, rate: 400, damage: 25, auto: true, max: 999 }
            ],
            lastFireTime: 0,
            isFiring: false
        };

        // --- THREE.JS SETUP ---
        const container = document.getElementById('gameCanvas');
        const scene = new THREE.Scene();
        scene.fog = new THREE.FogExp2(0x020202, 0.08); // Thick dark fog
        scene.background = new THREE.Color(0x000000);

        const camera = new THREE.PerspectiveCamera(70, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ antialias: false });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.shadowMap.enabled = true;
        renderer.shadowMap.type = THREE.PCFSoftShadowMap;
        container.appendChild(renderer.domElement);

        // --- LIGHTING ---
        const ambientLight = new THREE.AmbientLight(0xffffff); // Barely visible
        scene.add(ambientLight);
        
        // Player flashlight
        const flashLight = new THREE.SpotLight(0xfff0e0, 200, 160, Math.PI/2, 0.8, 2);
        flashLight.castShadow = true;
        scene.add(flashLight);
        const flashTarget = new THREE.Object3D();
        scene.add(flashTarget);
        flashLight.target = flashTarget;

        // --- TEXTURE GENERATION (2D onto 3D) ---
        function createCreepyTexture() {
            const canvas = document.createElement('canvas');
            canvas.width = 256; canvas.height = 256;
            const ctx = canvas.getContext('2d');
            
            // Base skin
            ctx.fillStyle = '#1a1818';
            ctx.fillRect(0, 0, 256, 256);
            
            // Hollow bleeding eyes
            ctx.fillStyle = '#000';
            ctx.beginPath(); ctx.ellipse(80, 100, 20, 30, 0, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.ellipse(176, 100, 20, 30, 0, 0, Math.PI*2); ctx.fill();
            
            // Blood streaks
            ctx.fillStyle = '#400';
            ctx.fillRect(70, 120, 15, 60);
            ctx.fillRect(85, 125, 5, 40);
            ctx.fillRect(166, 120, 15, 80);

            const tex = new THREE.CanvasTexture(canvas);
            tex.magFilter = THREE.NearestFilter; // Grungy look
            return tex;
        }
        const faceTex = createCreepyTexture();

        // --- PROCEDURAL ENVIRONMENT (Alleys) ---
        const walls = [];
        const mapSize = 16;
        const cellSize = 8;
        const wallMat1 = new THREE.MeshStandardMaterial({ color: 0x111111, roughness: 0.9 });
        const floorMat = new THREE.MeshStandardMaterial({ color: 0x0a0a0a, roughness: 1 });

        // Floor
        const floorGeo = new THREE.PlaneGeometry(mapSize * cellSize, mapSize * cellSize);
        const floor = new THREE.Mesh(floorGeo, floorMat);
        floor.rotation.x = -Math.PI / 2;
        floor.receiveShadow = true;
        scene.add(floor);

        // Generate Maze
        const boxGeo = new THREE.BoxGeometry(cellSize, 12, cellSize);
        for(let i = 0; i < mapSize; i++) {
            for(let j = 0; j < mapSize; j++) {
                if(Math.random() > 0.65 && (i !== mapSize/2 || j !== mapSize/2)) {
                    const wall = new THREE.Mesh(boxGeo, wallMat1);
                    wall.position.set((i - mapSize/2) * cellSize, 6, (j - mapSize/2) * cellSize);
                    wall.castShadow = true;
                    wall.receiveShadow = true;
                    scene.add(wall);
                    walls.push(wall);
                }
            }
        }

        // --- PLAYER SETUP ---
        const playerHeight = 4;
        camera.position.set(0, playerHeight, 0);
        const playerVelocity = new THREE.Vector3();
        const playerDirection = new THREE.Vector3();
        let euler = new THREE.Euler(0, 0, 0, 'YXZ');
        let bobbingTimer = 0;

        // Weapon Model
        const weaponGroup = new THREE.Group();
        const gunGeo = new THREE.BoxGeometry(0.3, 0.4, 1.5);
        const gunMat = new THREE.MeshStandardMaterial({ color: gameState.weapons[0].color, metalness: 0.8, roughness: 0.2 });
        const gunMesh = new THREE.Mesh(gunGeo, gunMat);
        gunMesh.position.set(0.6, -0.6, -1.2);
        gunMesh.castShadow = true;
        weaponGroup.add(gunMesh);
        camera.add(weaponGroup);
        scene.add(camera);

        // --- ENEMIES (Custom Hardcoded Primitives) ---
        const enemies = [];
        const bodyMat = new THREE.MeshStandardMaterial({ color: 0x1a1818, roughness: 0.8 });
        
        // Materials array for the head to put texture only on the front face
        const headMaterials = [
            bodyMat, bodyMat, bodyMat, bodyMat, 
            new THREE.MeshStandardMaterial({ map: faceTex, roughness: 0.8 }), // Front
            bodyMat
        ];

        function spawnEnemy() {
            const entity = new THREE.Group();
            
            // Head
            const head = new THREE.Mesh(new THREE.BoxGeometry(1.2, 1.4, 1.2), headMaterials);
            head.position.y = 4.7;
            head.castShadow = true;
            entity.add(head);

            // Torso
            const torso = new THREE.Mesh(new THREE.BoxGeometry(1.5, 3, 1), bodyMat);
            torso.position.y = 2.5;
            torso.castShadow = true;
            entity.add(torso);

            // Limbs
            const armGeo = new THREE.BoxGeometry(0.4, 3, 0.4);
            const lArm = new THREE.Mesh(armGeo, bodyMat);
            lArm.position.set(-0.95, 2.5, 0); lArm.castShadow = true;
            entity.add(lArm);
            
            const rArm = new THREE.Mesh(armGeo, bodyMat);
            rArm.position.set(0.95, 2.5, 0); rArm.castShadow = true;
            entity.add(rArm);

            const legGeo = new THREE.BoxGeometry(0.5, 3, 0.5);
            const lLeg = new THREE.Mesh(legGeo, bodyMat);
            lLeg.position.set(-0.4, -0.5, 0); lLeg.castShadow = true;
            entity.add(lLeg);
            
            const rLeg = new THREE.Mesh(legGeo, bodyMat);
            rLeg.position.set(0.4, -0.5, 0); rLeg.castShadow = true;
            entity.add(rLeg);

            // Spawn far away
            let angle = Math.random() * Math.PI * 2;
            let dist = 25 + Math.random() * 15;
            entity.position.set(Math.cos(angle)*dist, 2, Math.sin(angle)*dist);
            
            entity.userData = { health: 60, head, lArm, rArm, lLeg, rLeg, timeOffset: Math.random() * 10, attackCooldown: 0 };
            scene.add(entity);
            enemies.push(entity);
        }
        for(let i=0; i<4; i++) spawnEnemy();

        // --- PARTICLES ---
        const particles = [];
        function createImpact(position, color = 0xffaa00) {
            const geom = new THREE.BoxGeometry(0.1, 0.1, 0.1);
            const mat = new THREE.MeshBasicMaterial({color: color});
            for(let i=0; i<4; i++) {
                const p = new THREE.Mesh(geom, mat);
                p.position.copy(position);
                p.velocity = new THREE.Vector3((Math.random()-0.5)*15, Math.random()*10, (Math.random()-0.5)*15);
                p.life = 0.5;
                scene.add(p);
                particles.push(p);
            }
        }

        // --- CONTROLS & INPUT ---
        const input = { moveX: 0, moveY: 0 };
        const moveZone = document.getElementById('move-zone');
        const lookZone = document.getElementById('look-zone');
        const joyBase = document.getElementById('joystick-base');
        const joyStick = document.getElementById('joystick-stick');
        let moveTouchId = null, lookTouchId = null;
        let moveStartPos = { x: 0, y: 0 }, lookLastPos = { x: 0, y: 0 };
        const lookSensitivity = 0.004;

        moveZone.addEventListener('touchstart', (e) => {
            e.preventDefault();
            for(let t of e.changedTouches) {
                if(moveTouchId === null) {
                    moveTouchId = t.identifier;
                    moveStartPos = { x: t.clientX, y: t.clientY };
                    joyBase.style.display = 'block';
                    joyBase.style.left = (t.clientX - 60) + 'px';
                    joyBase.style.top = (t.clientY - 60) + 'px';
                    joyStick.style.transform = `translate(0px, 0px)`;
                }
            }
        });

        moveZone.addEventListener('touchmove', (e) => {
            e.preventDefault();
            for(let t of e.changedTouches) {
                if(t.identifier === moveTouchId) {
                    let dx = t.clientX - moveStartPos.x;
                    let dy = t.clientY - moveStartPos.y;
                    const dist = Math.sqrt(dx*dx + dy*dy);
                    const maxDist = 40;
                    if(dist > maxDist) { dx = (dx/dist)*maxDist; dy = (dy/dist)*maxDist; }
                    joyStick.style.transform = `translate(${dx}px, ${dy}px)`;
                    input.moveX = dx / maxDist;
                    input.moveY = dy / maxDist;
                }
            }
        });

        const endMove = (e) => {
            for(let t of e.changedTouches) {
                if(t.identifier === moveTouchId) {
                    moveTouchId = null; input.moveX = 0; input.moveY = 0; joyBase.style.display = 'none';
                }
            }
        };
        moveZone.addEventListener('touchend', endMove);
        moveZone.addEventListener('touchcancel', endMove);

        const handleLookStart = (t) => { lookTouchId = t.identifier; lookLastPos = { x: t.clientX, y: t.clientY }; };
        const handleLookMove = (t) => {
            let dx = t.clientX - lookLastPos.x; let dy = t.clientY - lookLastPos.y;
            lookLastPos = { x: t.clientX, y: t.clientY };
            euler.y -= dx * lookSensitivity * (gameState.isADS ? 0.4 : 1);
            euler.x -= dy * lookSensitivity * (gameState.isADS ? 0.4 : 1);
            euler.x = Math.max(-Math.PI/2.2, Math.min(Math.PI/2.2, euler.x));
            camera.quaternion.setFromEuler(euler);
        };

        lookZone.addEventListener('touchstart', (e) => { e.preventDefault(); for(let t of e.changedTouches) if(lookTouchId === null) handleLookStart(t); });
        lookZone.addEventListener('touchmove', (e) => { e.preventDefault(); for(let t of e.changedTouches) if(t.identifier === lookTouchId) handleLookMove(t); });
        const endLook = (e) => { for(let t of e.changedTouches) if(t.identifier === lookTouchId) lookTouchId = null; };
        lookZone.addEventListener('touchend', endLook);
        lookZone.addEventListener('touchcancel', endLook);

        // Buttons
        const btnFire = document.getElementById('btn-fire');
        let fireTouchId = null;
        btnFire.addEventListener('touchstart', (e) => {
            e.preventDefault(); gameState.isFiring = true;
            for(let t of e.changedTouches) { fireTouchId = t.identifier; if(lookTouchId === null) handleLookStart(t); }
        });
        btnFire.addEventListener('touchmove', (e) => { e.preventDefault(); for(let t of e.changedTouches) if(t.identifier === fireTouchId) handleLookMove(t); });
        const endFire = (e) => {
            gameState.isFiring = false;
            for(let t of e.changedTouches) {
                if(t.identifier === fireTouchId) { fireTouchId = null; if(lookTouchId === t.identifier) lookTouchId = null; }
            }
        };
        btnFire.addEventListener('touchend', endFire);
        btnFire.addEventListener('touchcancel', endFire);

        document.getElementById('btn-jump').addEventListener('touchstart', () => { if(camera.position.y <= playerHeight + 0.1) playerVelocity.y = 12; });
        
        document.getElementById('btn-flash').addEventListener('touchstart', () => {
            gameState.flashOn = !gameState.flashOn;
            flashLight.intensity = gameState.flashOn ? 200 : 0;
            document.getElementById('btn-flash').style.borderColor = gameState.flashOn ? "#fff" : "#555";
        });

        document.getElementById('btn-swap').addEventListener('touchstart', () => {
            gameState.weaponIndex = (gameState.weaponIndex + 1) % gameState.weapons.length;
            const w = gameState.weapons[gameState.weaponIndex];
            gunMat.color.setHex(w.color);
            gameState.maxAmmo = w.max; gameState.ammo = w.max;
            updateUI();
        });

        document.getElementById('btn-ads').addEventListener('touchstart', () => {
            gameState.isADS = !gameState.isADS;
            if(gameState.isADS) { gunMesh.position.set(0, -0.3, -1); camera.fov = 45; } 
            else { gunMesh.position.set(0.6, -0.6, -1.2); camera.fov = 70; }
            camera.updateProjectionMatrix();
        });

        document.getElementById('btn-reload').addEventListener('touchstart', () => {
            if(!gameState.isReloading && gameState.ammo < gameState.maxAmmo) {
                gameState.isReloading = true;
                gunMesh.rotation.x = -Math.PI/4;
                setTimeout(() => {
                    gameState.ammo = gameState.maxAmmo; gameState.isReloading = false; gunMesh.rotation.x = 0; updateUI();
                }, 1800);
            }
        });

        // --- SHOOTING LOGIC ---
        const raycaster = new THREE.Raycaster();
        const center = new THREE.Vector2(0, 0);

        function fireWeapon() {
            const weapon = gameState.weapons[gameState.weaponIndex];
            const now = performance.now();
            if (now - gameState.lastFireTime < weapon.rate || gameState.isReloading) return;
            if (weapon.max !== 999 && gameState.ammo <= 0) return;
            
            gameState.lastFireTime = now;
            if(weapon.max !== 999) gameState.ammo--;
            updateUI();

            // Recoil
            gunMesh.position.z += 0.3; gunMesh.rotation.x += 0.1;
            setTimeout(() => { gunMesh.position.z -= 0.3; gunMesh.rotation.x -= 0.1; }, 100);

            // Flash effect
            const originalInt = flashLight.intensity;
            flashLight.intensity = 500; setTimeout(() => flashLight.intensity = gameState.flashOn ? 200 : 0, 50);

            // Raycast
            raycaster.setFromCamera(center, camera);
            // Intersect walls and enemy GROUPS (recursive)
            const intersects = raycaster.intersectObjects([...enemies, ...walls], true);
            
            if (intersects.length > 0) {
                const hit = intersects[0];
                let isEnemy = false;
                let hitObj = hit.object;
                
                // Traverse up to find the group
                while(hitObj.parent && hitObj.parent.type === "Group") { hitObj = hitObj.parent; }
                
                if(enemies.includes(hitObj)) {
                    createImpact(hit.point, 0x880000); // Blood
                    hitObj.userData.health -= weapon.damage;
                    if (hitObj.userData.health <= 0) {
                        scene.remove(hitObj);
                        enemies.splice(enemies.indexOf(hitObj), 1);
                        setTimeout(spawnEnemy, 3000);
                    }
                } else {
                    createImpact(hit.point, 0xaaaaaa); // Sparks/Dust
                }
            }
        }

        // --- GAME LOOP ---
        const clock = new THREE.Clock();
        const uiAmmo = document.getElementById('ammo-display');
        const uiHealth = document.getElementById('health-fill');
        const uiBattery = document.getElementById('battery-fill');
        const dmgOverlay = document.getElementById('damage-overlay');
        const staticOverlay = document.getElementById('static-overlay');

        function updateUI() {
            uiAmmo.innerText = gameState.weapons[gameState.weaponIndex].max === 999 ? "â" : `${gameState.ammo}`;
            uiHealth.style.width = `${gameState.health}%`;
        }

        function takeDamage(amount) {
            gameState.health -= amount;
            dmgOverlay.style.opacity = 1; staticOverlay.style.opacity = 0.5;
            setTimeout(() => { dmgOverlay.style.opacity = 0; staticOverlay.style.opacity = 0; }, 200);
            updateUI();
            if(gameState.health <= 0) {
                document.body.innerHTML = '<h1 style="color:red; text-align:center; margin-top:50%; font-family:monospace;">THE FLESH CONSUMES.</h1>';
            }
        }

        function animate() {
            requestAnimationFrame(animate);
            const delta = Math.min(clock.getDelta(), 0.1);
            const time = clock.getElapsedTime();

            // Firing
            if(gameState.isFiring) {
                const w = gameState.weapons[gameState.weaponIndex];
                if(w.auto || performance.now() - gameState.lastFireTime > w.rate + 50) fireWeapon();
            }

            // Battery Management
            if(gameState.flashOn) {
                gameState.battery = Math.max(0, gameState.battery - delta * 2);
                if(gameState.battery === 0) { gameState.flashOn = false; flashLight.intensity = 0; document.getElementById('btn-flash').style.borderColor = "#555"; }
            } else {
                gameState.battery = Math.min(100, gameState.battery + delta * 5);
            }
            uiBattery.style.width = `${gameState.battery}%`;
            
            // Flicker if low
            if(gameState.flashOn && gameState.battery < 20 && Math.random() > 0.8) flashLight.intensity = Math.random() * 50;
            else if(gameState.flashOn) flashLight.intensity = 200;

            // Player Movement
            playerVelocity.x = 0; playerVelocity.z = 0;
            const speed = 8;
            if (input.moveY !== 0 || input.moveX !== 0) {
                playerDirection.set(input.moveX, 0, input.moveY).applyEuler(new THREE.Euler(0, euler.y, 0)).normalize();
                playerVelocity.x = playerDirection.x * speed;
                playerVelocity.z = playerDirection.z * speed;
                
                // Bobbing
                bobbingTimer += delta * 12;
                camera.position.y = playerHeight + Math.sin(bobbingTimer) * 0.15;
            } else {
                camera.position.y += (playerHeight - camera.position.y) * 10 * delta;
            }

            // Gravity
            playerVelocity.y -= 30 * delta;
            camera.position.addScaledVector(playerVelocity, delta);
            if (camera.position.y < playerHeight && playerVelocity.y <= 0) {
                playerVelocity.y = 0; camera.position.y = playerHeight;
            }

            // Wall Collision
            const pX = camera.position.x, pZ = camera.position.z;
            for(let w of walls) {
                const box = new THREE.Box3().setFromObject(w);
                if(pX > box.min.x - 0.5 && pX < box.max.x + 0.5 && pZ > box.min.z - 0.5 && pZ < box.max.z + 0.5) {
                    camera.position.x -= playerVelocity.x * delta * 1.5;
                    camera.position.z -= playerVelocity.z * delta * 1.5;
                }
            }

            // Flashlight Update
            flashLight.position.copy(camera.position);
            flashTarget.position.copy(camera.position).add(new THREE.Vector3(0,0,-1).applyQuaternion(camera.quaternion));

            // Enemy AI & Animations
            enemies.forEach(enemy => {
                const dist = enemy.position.distanceTo(camera.position);
                enemy.lookAt(camera.position.x, enemy.position.y, camera.position.z);
                
                // Mechanics: Move fast in dark, crawl in light
                let isSeen = false;
                if(gameState.flashOn) {
                    const toEnemy = enemy.position.clone().sub(camera.position).normalize();
                    const camDir = new THREE.Vector3(0,0,-1).applyQuaternion(camera.quaternion);
                    if(camDir.dot(toEnemy) > 0.8) isSeen = true; // In spotlight
                }
                
                const moveSpeed = isSeen ? 0 : 3; // Terrifying speed in darkness
                const animSpeed = isSeen ? 0 : 12;

                if (dist > 3) {
                    enemy.translateZ(moveSpeed * delta);
                    
                    // Procedural Animation
                    const tOffset = enemy.userData.timeOffset;
                    enemy.userData.lArm.rotation.x = Math.sin(time * animSpeed + tOffset) * 0.8;
                    enemy.userData.rArm.rotation.x = -Math.sin(time * animSpeed + tOffset) * 0.8;
                    enemy.userData.lLeg.rotation.x = -Math.sin(time * animSpeed + tOffset) * 0.6;
                    enemy.userData.rLeg.rotation.x = Math.sin(time * animSpeed + tOffset) * 0.6;
                    
                    // Creepy head twitch
                    if(Math.random() > 0.95) enemy.userData.head.rotation.y = (Math.random()-0.5) * 0.5;
                    if(Math.random() > 0.95) enemy.userData.head.rotation.z = (Math.random()-0.5) * 0.3;
                    
                } else {
                    // Attack
                    if (time - enemy.userData.attackCooldown > 1.5) {
                        enemy.userData.attackCooldown = time;
                        takeDamage(20);
                        // Attack animation
                        enemy.userData.lArm.rotation.x = -Math.PI/2;
                        enemy.userData.rArm.rotation.x = -Math.PI/2;
                    }
                }
            });

            // Particles
            for(let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.position.addScaledVector(p.velocity, delta);
                p.velocity.y -= 20 * delta;
                p.life -= delta;
                if(p.life <= 0) { scene.remove(p); particles.splice(i, 1); }
            }

            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        updateUI();
        animate();
    </script>
