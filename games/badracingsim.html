function clamp(v, a, b) {
    return Math.max(a, Math.min(b, v));
}

function getTorque(rpm) {
    const x = clamp(rpm / C.maxRPM, 0, 1);
    let factor;

    if (x < 0.12) {
        factor = 0.45 + (x / 0.12) * 0.20;
    } else if (x < 0.50) {
        factor = 0.65 + ((x - 0.12) / 0.38) * 0.35;
    } else if (x < 0.82) {
        factor = 1.00 - ((x - 0.50) / 0.32) * 0.20;
    } else {
        factor = 0.80 - ((x - 0.82) / 0.18) * 0.45;
    }

    return C.maxTorque * clamp(factor, 0.35, 1.0);
}

function tireGrip(load, mu) {
    const L = Math.max(0, load);
    return mu * Math.pow(L, 0.92);
}

function computeAxleLoads(ax, ay) {
    const weight = C.mass * 9.81;

    const staticFront = weight * (C.lr / C.wheelbase);
    const staticRear  = weight * (C.lf / C.wheelbase);

    const longTransfer = (C.mass * ax * C.cgHeight) / C.wheelbase;
    const latTransfer = (C.mass * ay * C.cgHeight) / C.track;

    let front = staticFront - longTransfer;
    let rear = staticRear + longTransfer;

    front = Math.max(0, front);
    rear = Math.max(0, rear);

    let frontLeft = Math.max(0, front * 0.5 - latTransfer * 0.5);
    let frontRight = Math.max(0, front * 0.5 + latTransfer * 0.5);
    let rearLeft = Math.max(0, rear * 0.5 - latTransfer * 0.5);
    let rearRight = Math.max(0, rear * 0.5 + latTransfer * 0.5);

    return { frontLeft, frontRight, rearLeft, rearRight };
}

function updatePowertrain(dt) {
    const wheelSpeedRad = Math.abs(S.vx) / wheelRadius;
    const currentRatio = C.gears[S.gear] * C.finalDrive;

    let targetRPM = wheelSpeedRad * currentRatio * (60 / (2 * Math.PI));
    if (targetRPM < C.idleRPM) targetRPM = C.idleRPM;

    S.rpm += (targetRPM - S.rpm) * 10 * dt;

    if (S.shiftTimer > 0) {
        S.shiftTimer -= dt;
    } else {
        if (S.rpm > C.shiftUpRPM && S.gear < C.gears.length - 1) {
            S.gear++;
            S.shiftTimer = 0.22;
        } else if (S.rpm < C.shiftDownRPM && S.gear > 0 && Math.abs(S.vx) > 1.2) {
            S.gear--;
            S.shiftTimer = 0.22;
        }
    }

    let driveRequest = 0;
    if (inputs.throttle > 0) {
        const engineTorque = getTorque(S.rpm) * inputs.throttle;
        driveRequest = (engineTorque * currentRatio) / wheelRadius;
    }

    return driveRequest;
}

function updateSuspension(dt, ax, ay) {
    const targetPitch = -ax * 0.006;
    const targetRoll = ay * -0.008;
    S.pitch += (targetPitch - S.pitch) * 10 * dt;
    S.roll += (targetRoll - S.roll) * 10 * dt;
    carBody.rotation.x = S.pitch;
    carBody.rotation.z = S.roll;
}

function ccw(ax, az, bx, bz, cx, cz) {
    return (cz - az) * (bx - ax) > (bz - az) * (cx - ax);
}

function segmentsIntersect(ax, az, bx, bz, cx, cz, dx, dz) {
    return ccw(ax, az, cx, cz, dx, dz) !== ccw(bx, bz, cx, cz, dx, dz) &&
           ccw(ax, az, bx, bz, cx, cz) !== ccw(ax, az, bx, bz, dx, dz);
}

const lapState = {

function formatLapTime(ms) {
    const total = Math.max(0, ms);
    const minutes = Math.floor(total / 60000);
    const seconds = Math.floor((total % 60000) / 1000);
    const millis = Math.floor(total % 1000);
    return `${minutes}:${String(seconds).padStart(2, '0')}.${String(millis).padStart(3, '0')}`;
}

function updateLapSystem(prevX, prevZ, currX, currZ, dt) {
    lapState.lapTimeMs += dt * 1000;

    const gate = checkpoints[lapState.nextCheckpoint];
    if (!gate) return;

    if (segmentsIntersect(prevX, prevZ, currX, currZ, gate.a.x, gate.a.z, gate.b.x, gate.b.z)) {
        if (lapState.nextCheckpoint === 0) {
            if (lapState.lapTimeMs > 5000) {
                lapState.lapCount++;
                if (lapState.bestMs === null || lapState.lapTimeMs < lapState.bestMs) {
                    lapState.bestMs = lapState.lapTimeMs;
                }
            }
            lapState.lapTimeMs = 0;
            lapState.nextCheckpoint = 1;
        } else {
            lapState.nextCheckpoint++;
            if (lapState.nextCheckpoint >= checkpoints.length) {
                lapState.nextCheckpoint = 0;
            }
        }
    }
}

function updateSlipFX(dt) {
    for (let i = skidMarks.length - 1; i >= 0; i--) {
        const s = skidMarks[i];
        s.life -= dt;
        const t = clamp(s.life / s.maxLife, 0, 1);
        s.mesh.material.opacity = 0.45 * t;
        if (s.life <= 0) {
            scene.remove(s.mesh);
            s.mesh.geometry.dispose();
            s.mesh.material.dispose();
            skidMarks.splice(i, 1);
        }
    }

    for (let i = smokePuffs.length - 1; i >= 0; i--) {
        const p = smokePuffs[i];
        p.life -= dt;
        p.sprite.position.addScaledVector(p.vel, dt);
        p.vel.y += 0.35 * dt;

        const t = clamp(p.life / p.maxLife, 0, 1);
        const size = p.baseScale * (1 + (1 - t) * 1.9);
        p.sprite.scale.set(size, size, 1);
        p.sprite.material.opacity = 0.42 * t;

        if (p.life <= 0) {
            scene.remove(p.sprite);
            p.sprite.material.dispose();
            smokePuffs.splice(i, 1);
        }
    }

    const slip = S.slipIntensity;
    const speed = Math.abs(S.vx);

    for (let i = 0; i < 2; i++) {
        const wheelIndex = i + 2;
        const fx = rearVFX[i];
        const wp = new THREE.Vector3();
        wheels[wheelIndex].group.getWorldPosition(wp);
        wp.y = 0.113;

        if (slip > 0.08 && speed > 0.4) {
            if (fx.lastPos && fx.lastPos.distanceTo(wp) > 0.04) {
                spawnSkidMark(
                    fx.lastPos,
                    wp,
                    0.15 + slip * 0.14,
                    0.18 + slip * 0.22
                );
                fx.lastPos.copy(wp);
            } else if (!fx.lastPos) {
                fx.lastPos = wp.clone();
            }

            fx.smokeAccum += dt * (3 + slip * 18);
            while (fx.smokeAccum >= 1) {
                spawnSmoke(wp, slip, i === 0 ? -1 : 1);
                fx.smokeAccum -= 1;
            }
        } else {
            fx.lastPos = null;
            fx.smokeAccum = 0;
        }
    }
}

function updatePhysics(dt) {
    const prevX = S.x;
    const prevZ = S.z;

    const speed = Math.abs(S.vx);

    const steerFactor = 1.0 / (1.0 + speed * 0.0075);
    S.steerAngle += (inputs.steer * C.maxSteer * steerFactor - S.steerAngle) * 8 * dt;
    wheels.forEach(w => { if (w.isFront) w.group.rotation.y = S.steerAngle; });

    const driveRequest = updatePowertrain(dt);

    const signVel = Math.abs(S.vx) > 0.15 ? Math.sign(S.vx) : 0;
    const brakeForce = 9000 * inputs.brake;

    const loads = computeAxleLoads(S.ax, S.ay);

    const frontGrip = tireGrip(loads.frontLeft, C.mu) + tireGrip(loads.frontRight, C.mu);
    const rearGrip  = tireGrip(loads.rearLeft, C.mu) + tireGrip(loads.rearRight, C.mu);

    const vg = Math.max(speed, 1.0);

    const alphaF = S.steerAngle - Math.atan2(S.vy + C.lf * S.r, vg);
    const alphaR = -Math.atan2(S.vy - C.lr * S.r, vg);

    let Fyf = Math.tanh((C.mass * 6.0 * alphaF) / Math.max(frontGrip, 1)) * frontGrip;
    let Fyr = Math.tanh((C.mass * 6.0 * alphaR) / Math.max(rearGrip, 1)) * rearGrip;

    let FxFrontRaw = -brakeForce * 0.65 * signVel;
    let FxRearRaw = driveRequest - brakeForce * 0.35 * signVel;

    const frontLongCap = Math.sqrt(Math.max(0, frontGrip * frontGrip - Fyf * Fyf));
    const rearLongCap  = Math.sqrt(Math.max(0, rearGrip  * rearGrip  - Fyr * Fyr));

    const FxFront = clamp(FxFrontRaw, -frontLongCap, frontLongCap);
    const FxRear  = clamp(FxRearRaw,  -rearLongCap,  rearLongCap);

    const driveSlip = Math.max(0, driveRequest - rearLongCap) / Math.max(rearLongCap, 1);
    const brakeSlip = Math.max(0, Math.abs(FxRearRaw) - rearLongCap) / Math.max(rearLongCap, 1);
    S.slipIntensity = clamp(Math.max(driveSlip, brakeSlip), 0, 1);

    S.spinning = inputs.throttle > 0 && driveRequest > rearLongCap + 1;

    if (S.spinning) {
        S.rpm += 1800 * dt;
    }

    const Fx = FxFront + FxRear - (C.drag * S.vx * speed) - (C.rollRes * signVel);
    const ax = Fx / C.mass + S.vy * S.r;
    const ay = (Fyf + Fyr) / C.mass - S.vx * S.r;
    const ar = (C.lf * Fyf - C.lr * Fyr) / C.Iz;

    S.vx += ax * dt;
    S.vy += ay * dt;
    S.r  += ar * dt;

    S.vy *= (1.0 - 1.6 * dt);
    S.r  *= (1.0 - 0.6 * dt);

    S.ax = ax;
    S.ay = ay;

    updateSuspension(dt, ax, ay);

    S.heading += S.r * dt;

    const vwx = S.vx * Math.sin(S.heading) + S.vy * Math.cos(S.heading);
    const vwz = S.vx * Math.cos(S.heading) - S.vy * Math.sin(S.heading);
    const nX = S.x + vwx * dt;
    const nZ = S.z + vwz * dt;

    S.x = nX;
    S.z = nZ;

    updateLapSystem(prevX, prevZ, S.x, S.z, dt);

    carRoot.position.set(S.x, 0, S.z);
    carRoot.rotation.y = S.heading;
}

function animate() {
    requestAnimationFrame(animate);
    const dt = Math.min(clock.getDelta(), 0.033);

    updatePhysics(dt);

    for (const w of wheels) {
        const localX = w.group.position.x;
        const wheelContactSpeed = S.vx - S.r * localX;
        const wheelSpin = (wheelContactSpeed / wheelRadius) * dt;
        const spinExtra = (!w.isFront && S.spinning) ? 2.5 : 1.0;
        w.mesh.rotation.x += wheelSpin * spinExtra;
    }

    carRoot.updateMatrixWorld(true);
    updateSlipFX(dt);

    speedLabel.innerText = Math.round(Math.abs(S.vx) * 3.6) + ' km/h';

    const rpmAngle = -90 + (Math.min(S.rpm, C.maxRPM) / C.maxRPM) * 180;
    tachoNeedle.style.transform = `translateX(-50%) rotate(${rpmAngle}deg)`;

    gearLabel.innerText = 'D' + (S.gear + 1);
    tractionWarn.style.opacity = S.spinning ? '1' : '0';

    lapCountLabel.innerText = 'LAPS ' + lapState.lapCount;
    lapTimeLabel.innerText = 'LAP ' + formatLapTime(lapState.lapTimeMs);
    bestTimeLabel.innerText = 'BEST ' + (lapState.bestMs === null ? '--:--.---' : formatLapTime(lapState.bestMs));

    const camOffset = new THREE.Vector3(0, 3.5, -8).applyAxisAngle(new THREE.Vector3(0,1,0), S.heading);
    camera.position.lerp(carRoot.position.clone().add(camOffset), 6 * dt);
    camera.lookAt(carRoot.position.clone().add(new THREE.Vector3(0,1.2,0)));

    renderer.render(scene, camera);
}
animate();
