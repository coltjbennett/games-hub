    function recalculateStats() {
        let gps = 0;
        let clickPwr = 1;
        let tempMults = { click: 1, global: 1 };
        buildings.forEach(b => tempMults[b.id] = 1);

        let boughtUpgs = 0;
        upgrades.filter(u => u.bought).forEach(u => {
            boughtUpgs++;
            if (tempMults[u.target] !== undefined) tempMults[u.target] *= u.mult;
        });

        let spellCostMult = 1;
        let relicDropMult = 1;
        let goldenElixirMult = 1;

        game.activeCollectibles.forEach(cid => {
            const coll = collectibles.find(c => c.id === cid);
            if (!coll) return;
            if (coll.type === 'unit' && tempMults[coll.effectTarget] !== undefined) tempMults[coll.effectTarget] *= coll.effectMult;
            if (coll.type === 'click') tempMults.click *= coll.effectMult;
            if (coll.type === 'global') tempMults.global *= coll.effectMult;
            if (coll.type === 'spell') spellCostMult *= coll.effectMult;
        });

        trophyMultiplier = 1 + (game.achievements.length * 0.015);
        tempMults.global *= trophyMultiplier;

        if (game.proceduralResearchLevel > 0) {
            tempMults.global *= (1 + game.proceduralResearchLevel * 0.015);
            tempMults.click *= (1 + game.proceduralResearchLevel * 0.003);
        }

        rbUpgrades.forEach(r => {
            if (!r.bought) return;
            if (r.effectType === 'globalMult') tempMults.global *= (1 + r.effectValue);
            if (r.effectType === 'clickMult') tempMults.click *= (1 + r.effectValue);
            if (r.effectType === 'buildingMult' && tempMults[r.effectTarget] !== undefined) tempMults[r.effectTarget] *= (1 + r.effectValue);
            if (r.effectType === 'spellCost') spellCostMult *= (1 - r.effectValue);
            if (r.effectType === 'relicChance') relicDropMult *= (1 + r.effectValue);
            if (r.effectType === 'goldenElixir') goldenElixirMult *= (1 + r.effectValue);
        });

        buildings.forEach(b => {
            b.mult = tempMults[b.id];
            gps += (b.count * b.baseGps * b.mult);
        });

        if (researches[2].bought) gps *= (1 + (boughtUpgs * 0.01));
        if (researches[3].bought) gps *= 1.20;

        gps *= tempMults.global;

        if (game.spellActive.blessing > 0) gps *= 3;
        if (game.spellActive.overdrive > 0) gps *= 1.5;

        if (rbUpgrades[0].bought) gps *= 2;
        if (rbUpgrades[1].bought) tempMults.click *= 2;

        clickPwr = (1 * tempMults.click) + (gps * 0.01);
        if (clickPwr < 1) clickPwr = 1;

        if (game.spellActive.frenzy > 0) clickPwr *= 10;
        if (game.spellActive.overdrive > 0) clickPwr *= 5;

        cachedGps = gps;
        cachedClickPower = clickPwr;
        game.spellCostMult = spellCostMult;
        game.relicDropMult = relicDropMult;
        game.goldenElixirMult = goldenElixirMult;
    }

        const unowned = collectibles.filter(c => !game.collectibles.includes(c.id));

    // --- MAGIC SYSTEM ---
    function castSpell(id) {
        const spell = gameSpells.find(s => s.id === id);
        if(!spell) return;
        if(game.spellCooldowns[id] > 0 || game.spellActive[id] > 0) return;

        const cost = getSpellCost(spell);
        if(spell.costType === 'goblins' && game.goblins < cost) return;
        if(spell.costType === 'rp' && game.rp < cost) return;

        if(spell.costType === 'goblins') game.goblins -= cost;
        else game.rp -= cost;

        if(id === 'frenzy') {
            game.spellActive.frenzy = spell.durMax;
            recalculateStats();
        } else if (id === 'blessing') {
            game.spellActive.blessing = spell.durMax;
            recalculateStats();
        } else if (id === 'overdrive') {
            game.spellActive.overdrive = spell.durMax;
            recalculateStats();
        } else if (id === 'warp') {
            const reward = cachedGps * 300;
            game.goblins += reward; game.totalGoblins += reward;
            alert(`Time Warp successful! Gained ${format(reward)} Goblins instantly.`);
        } else if (id === 'rush' || id === 'summon') {
            const count = id === 'rush' ? 3 : 5;
            for(let i = 0; i < count; i++) setTimeout(spawnGoldenElixir, i * 1500);
        } else if (id === 'harvest') {
            const reward = cachedGps * 900;
            game.goblins += reward; game.totalGoblins += reward;
            alert(`Harvest Moon successful! Gained ${format(reward)} Goblins instantly.`);
        } else if (id === 'clarity') {
            gameSpells.forEach(s => {
                game.spellCooldowns[s.id] = Math.max(0, game.spellCooldowns[s.id] - 30);
            });
            alert('Arcane Clarity rippled through the hut. Cooldowns were reduced!');
        }

        game.spellCooldowns[id] = spell.cdMax;
        game.spellsCast++;
        checkAchievements();
        renderSpells();
        updateUI(true);
    }

    function sacrificeGoblins() {
        const cost = getRpCost();
        if (game.goblins >= cost) {
            game.goblins -= cost;
            game.rp++;
            game.totalRpThisRebirth++;
            updateUI(true);
            renderResearch();

            const btn = document.getElementById('btn-sacrifice');
            btn.style.transform = 'scale(0.9)';
            setTimeout(() => btn.style.transform = 'scale(1)', 100);

            if (canRebirth()) {
                document.getElementById('rebirth-btn').style.display = 'block';
            }
        }
    }

    function buyResearch(i) {
        const r = researches[i];
        if (!r.bought && game.rp >= r.cost) {
            game.rp -= r.cost;
            r.bought = true;
            recalculateStats();
            renderResearch();
            updateUI(true);
        }
    }

    function buyProceduralResearch() {
        if (!allMainResearchBought()) return;
        const cost = getProceduralResearchCost();
        if (game.rp >= cost) {
            game.rp -= cost;
            game.proceduralResearchLevel++;
            recalculateStats();
            renderResearch();
            updateUI(true);
        }
    }

    function performRebirth() {
        if (!canRebirth()) return alert("You need 20 RP in the bank and all 4 main research upgrades before rebirthing.");
        const gain = getRebirthPointGain();
        if (confirm(`Rebirth? You will lose all Goblins, Buildings, Upgrades, and RP in exchange for ${gain} Rebirth Point${gain === 1 ? '' : 's'}!`)) {
            game.rbPoints += gain;
            game.rebirths++;

            game.goblins = 0; game.rp = 0; game.totalRpThisRebirth = 0;
            game.proceduralResearchLevel = 0;
            buildings.forEach(b => b.count = 0);
            upgrades.forEach(u => u.bought = false);
            researches.forEach(r => r.bought = false);

            wanderingGoblins.forEach(g => g.el.remove());
            wanderingGoblins = []; visualCounts = {};

            recalculateStats();
            initUI();
            updateUI(true);
            checkAchievements();
            closeModal('rebirth-modal');
        }
    }

    function buyRbUpgrade(i) {
        const r = rbUpgrades[i];

            game.rbPoints -= r.cost;

    function addVisualGoblin(spriteClass) {
        if (!game.motionEnabled) return;
        if (!visualCounts[spriteClass]) visualCounts[spriteClass] = 0;
        if (visualCounts[spriteClass] >= 7) return;

        const dot = document.createElement('div');
        dot.className = 'mini-goblin';
        dot.innerHTML = getSpriteHTML(spriteClass);
        
        const w = arenaDisp.clientWidth || 300;
        const h = arenaDisp.clientHeight || 300;
        const startX = Math.random() * Math.max(10, (w - 40));
        const startY = Math.random() * Math.max(10, (h - 40));
        
        dot.style.left = startX + 'px';
        dot.style.top = startY + 'px';
        dot.style.animationDelay = `${Math.random()}s`;
        
        arenaDisp.appendChild(dot);
        
        wanderingGoblins.push({
            el: dot, x: startX, y: startY,
            vx: (Math.random() - 0.5) * 80,
            vy: (Math.random() - 0.5) * 80
        });
        visualCounts[spriteClass]++;
    }

        floater.style.left = (e.clientX || arenaDisp.clientWidth/2) + 'px';
        floater.style.top = (e.clientY || arenaDisp.clientHeight/2) + 'px';
        document.body.appendChild(floater);
        setTimeout(() => floater.remove(), 2000);
        updateUI(true);
    }

    function updateUI(forceDOM = false) {
        countDisp.innerText = format(game.goblins);
        document.getElementById('rp-count').innerText = game.rp;
        document.getElementById('rb-count').innerText = game.rbPoints;
        document.getElementById('trophy-bonus-text').innerText = (game.achievements.length * 1.5).toFixed(1);

        const rpCost = getRpCost();
        const sacBtn = document.getElementById('btn-sacrifice');
        if (game.goblins >= rpCost) {
            sacBtn.style.opacity = 1; sacBtn.style.cursor = 'pointer';
        } else {
            sacBtn.style.opacity = 0.5; sacBtn.style.cursor = 'not-allowed';
        }

        document.getElementById('rebirth-btn').style.display = canRebirth() ? 'block' : 'none';

        if (forceDOM) {
            gpsDisp.innerText = format(cachedGps) + ' GPS';
            buildings.forEach((b, i) => {
                const el = document.getElementById(`bld-${i}`);
                if (!el) return;
                const cost = getCost(b);
                if (game.goblins >= cost) el.classList.remove('disabled');
                else el.classList.add('disabled');

                document.getElementById(`bld-cost-${i}`).innerText = format(cost);
                document.getElementById(`bld-count-${i}`).innerText = b.count;
                document.getElementById(`bld-gps-${i}`).innerText = `+${format(b.baseGps * b.mult)} GPS`;
            });
            upgrades.forEach((u, i) => {
                const el = document.getElementById(`upg-${i}`);
                if (!el) return;
                if (game.goblins >= u.cost) el.classList.remove('disabled');
                else el.classList.add('disabled');
            });
            researches.forEach((r, i) => {
                const el = document.getElementById(`res-${i}`);
                if (!el || r.bought) return;
                if (game.rp >= r.cost) el.classList.remove('disabled');
                else el.classList.add('disabled');
            });
            if (allMainResearchBought()) {
                const el = document.getElementById('proc-research-btn');
                if (el) {
                    if (game.rp >= getProceduralResearchCost()) el.classList.remove('disabled');
                    else el.classList.add('disabled');
                }
            }
            rbUpgrades.forEach((r, i) => {
                const el = document.getElementById(`rbupg-${i}`);
                if (!el || r.bought) return;
                if (game.rbPoints >= r.cost) el.classList.remove('disabled');
                else el.classList.add('disabled');
            });

            if(document.getElementById('tab-magic').classList.contains('active')) {
                gameSpells.forEach(s => {
                    const el = document.getElementById(`spl-${s.id}`);
                    if(!el) return;
                    const cost = getSpellCost(s);
                    const canAfford = (s.costType === 'goblins' && game.goblins >= cost) || (s.costType === 'rp' && game.rp >= cost);
                    const ready = game.spellCooldowns[s.id] <= 0 && game.spellActive[s.id] <= 0;

                    if(canAfford && ready) el.classList.remove('disabled');
                    else el.classList.add('disabled');

                    if(ready) document.getElementById(`spl-txt-${s.id}`).innerText = `${format(cost)} ${s.costType === 'rp' ? 'RP' : 'Gobs'}`;
                });
            }
        }
    }

    function updateNews() {
        let currentNews = newsQuotes[0].text;
        for (let i = newsQuotes.length - 1; i >= 0; i--) {
            if (game.totalGoblins >= newsQuotes[i].thres) {
                currentNews = newsQuotes[i].text;
                break;
            }
        }
        document.getElementById('news-ticker').innerText = currentNews;
    }

    let lastTime = Date.now();
    setInterval(() => {
        const now = Date.now();
        const delta = (now - lastTime) / 1000;
        lastTime = now;

        // Magic Timers
        let spellStateChanged = false;
        gameSpells.forEach(s => {
            if(game.spellActive[s.id] > 0) {
                game.spellActive[s.id] -= delta;
                if(game.spellActive[s.id] <= 0) {
                    game.spellActive[s.id] = 0;
                    spellStateChanged = true;
                    recalculateStats();
                }
            }
            if(game.spellCooldowns[s.id] > 0) {
                game.spellCooldowns[s.id] -= delta;
                if(game.spellCooldowns[s.id] <= 0) {
                    game.spellCooldowns[s.id] = 0;
                    spellStateChanged = true;
                }
            }
        });
        
        if (cachedGps > 0) {
            const gain = cachedGps * delta;
            game.goblins += gain;
            game.totalGoblins += gain;
            updateUI(false); 
        }

        if(spellStateChanged && document.getElementById('tab-magic').classList.contains('active')) {
            renderSpells();
            updateUI(true);
        } else if (document.getElementById('tab-magic').classList.contains('active')) {
            // Live update bars
            gameSpells.forEach(s => {
                const cdBar = document.getElementById(`spl-cd-${s.id}`);
                const actDiv = document.getElementById(`spl-act-${s.id}`);
                const txt = document.getElementById(`spl-txt-${s.id}`);
                if(cdBar && game.spellCooldowns[s.id] > 0) {
                    cdBar.style.width = `${(game.spellCooldowns[s.id]/s.cdMax)*100}%`;
                    if(txt) txt.innerText = `${Math.ceil(game.spellCooldowns[s.id])}s CD`;
                }
                if(actDiv && game.spellActive[s.id] > 0) {
                    if(txt) txt.innerText = `${Math.ceil(game.spellActive[s.id])}s`;
                }
            });
        }

        if (game.motionEnabled && wanderingGoblins.length > 0) {
            const w = arenaDisp.clientWidth;
            const h = arenaDisp.clientHeight;
            if (w > 0 && h > 0) {
                const maxX = w - 40; 
                const maxY = h - 40;
                wanderingGoblins.forEach(g => {
                    g.x += g.vx * delta; g.y += g.vy * delta;
                    if (g.x <= 0) { g.x = 0; g.vx *= -1; }
                    if (g.x >= maxX) { g.x = maxX; g.vx *= -1; }
                    if (g.y <= 0) { g.y = 0; g.vy *= -1; }
                    if (g.y >= maxY) { g.y = maxY; g.vy *= -1; }
                    g.el.style.left = g.x + 'px'; g.el.style.top = g.y + 'px';
                });
            }
        }
    }, 33);

    setInterval(() => updateUI(true), 250);
    setInterval(updateNews, 10000);
    setInterval(() => {
        const chance = researches[1].bought ? 0.02 : 0.01;
        if (Math.random() < chance) spawnGoldenElixir(); 
    }, 1000);

    // --- SAVE / LOAD ---
    function saveGame() {
        const saveData = {
            game: game,
            bcounts: buildings.map(b => b.count),
            ubought: upgrades.map(u => u.bought),
            rbought: researches.map(r => r.bought),
            rbbought: rbUpgrades.map(r => r.bought),
            lastTimeSaved: Date.now()
        };
        localStorage.setItem('goblinSaveData_v6', JSON.stringify(saveData));

        const notify = document.getElementById('save-notify');
        notify.style.opacity = 1;
        setTimeout(() => notify.style.opacity = 0, 2000);
    }

    function loadGame() {
        const saved = localStorage.getItem('goblinSaveData_v6');
        let offlineEarnings = 0; let offlineMs = 0;

        if (saved) {
            try {
                const data = JSON.parse(saved);
                game = { ...game, ...data.game };
                if(game.rebirths === undefined) game.rebirths = 0;
                if(game.rbPoints === undefined) game.rbPoints = 0;
                if(game.totalRpThisRebirth === undefined) game.totalRpThisRebirth = 0;
                if(game.collectibles === undefined) game.collectibles = [];
                if(game.activeCollectibles === undefined) game.activeCollectibles = [];
                if(game.achievements === undefined) game.achievements = [];
                if(game.spellsCast === undefined) game.spellsCast = 0;
                if(game.proceduralResearchLevel === undefined) game.proceduralResearchLevel = 0;
                if(game.spellCooldowns === undefined) game.spellCooldowns = {};
                if(game.spellActive === undefined) game.spellActive = {};
                game.spellCooldowns = { frenzy: 0, warp: 0, rush: 0, blessing: 0, overdrive: 0, summon: 0, harvest: 0, clarity: 0, ...(game.spellCooldowns || {}) };
                game.spellActive = { frenzy: 0, blessing: 0, overdrive: 0, ...(game.spellActive || {}) };

                if (data.bcounts) buildings.forEach((b, i) => b.count = data.bcounts[i] || 0);
                if (data.ubought) upgrades.forEach((u, i) => u.bought = data.ubought[i] || false);
                if (data.rbought) researches.forEach((r, i) => r.bought = data.rbought[i] || false);
                if (data.rbbought) rbUpgrades.forEach((r, i) => r.bought = data.rbbought[i] || false);

                recalculateStats();
                if (data.lastTimeSaved && cachedGps > 0) {
                    const elapsedMs = Date.now() - data.lastTimeSaved;
                    if (elapsedMs > 60000) {
                        const twelveHoursMs = 12 * 60 * 60 * 1000;
                        offlineMs = Math.min(elapsedMs, twelveHoursMs);
                        const offlineSeconds = offlineMs / 1000;
                        offlineEarnings = offlineSeconds * cachedGps * 0.25;
                    }
                }

                visualCounts = {};
                if(game.motionEnabled) {
                    buildings.forEach(b => {
                        let spawnAmt = Math.min(b.count, 7);
                        for(let i = 0; i < spawnAmt; i++) addVisualGoblin(b.sprite);
                    });
                }
            } catch (e) {
                console.error("Save file corrupted, starting fresh.");
            }
        }
        recalculateStats();
        initUI();
        if (offlineEarnings > 0) {
            game.goblins += offlineEarnings;
            game.totalGoblins += offlineEarnings;
            document.getElementById('offline-gains').innerText = format(offlineEarnings);
            document.getElementById('offline-time-text').innerText = formatTime(offlineMs) + '/12:00:00 max';
            document.getElementById('offline-modal').style.display = 'flex';
        }
        updateUI(true);
        updateNews();
        checkAchievements();
    }

    setInterval(saveGame, 15000);
    loadGame();
    setInterval(saveGame, 15000);
    loadGame();
</script>
